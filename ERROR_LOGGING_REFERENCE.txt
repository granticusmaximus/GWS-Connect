## 💻 Usage

### Route Handler:
```javascript
import { logError } from '../utils/logger.js';

try {
  // code
} catch (error) {
  await logError(error, {
    file: 'routes/file.js',
    function: 'POST /api/endpoint',
    operation: 'What you were doing',
    userId: req.user?.id,
  });
}
```

### Socket Handler:
```javascript
import { logSocketError } from '../utils/logger.js';

socket.on('event', async (data) => {
  try {
    // code
  } catch (error) {
    await logSocketError(error, {
      file: 'index.js',
      socketId: socket.id,
      userId: socket.user.id,
      eventName: 'event',
      operation: 'What you were doing',
    });
  }
});
```
